#### Introduction 
This guide is for developers looking to get started with the WetroCloud API quickly. Using simple cURL commands, you’ll learn how to:

- Create a collection
- Insert resources into the collection
- Query collection for answers
- Chat with collection for conversations
- Categorize resources
- Remove resources
- Delete Collection

**Prerequisites:**
- An API key from the [Wetrocloud Console](https://console.wetrocloud.com)

By the end, you’ll have queried your first resource using Wetrocloud API.

  <Step title="Create a Collection">
    First, create a `collection_id` to group your resources.
    <CodeGroup>
```python Python
from wetro import Wetrocloud

# Access Modules
client = Wetrocloud(api_key="your_api_key")
rag_client = client.rag

# Create a collection
collection_id = rag_client.create_collection(
  collection_id="my_unique_collection_id"
)

print(collection_id)
```
  
```javascript Javascript
import Wetrocloud from "wetro-sdk";

const wetrocloud = new Wetrocloud();

const collection_id = 'unique_id'; //this is optional if not specified one would be generated for you.
const response = await wetrocloud.createCollection({ collection_id });

console.log("Creating Collection", response);
```

```bash cURL
curl --request POST \
  --url https://api.wetrocloud.com/v1/collection/create/ \
  --header 'Authorization: Token <api-key>'
  --form 'collection_id=<collection_id>' \
```
    </CodeGroup>
  </Step>

  <Step title="Insert a Resource into the Collection">
    Add a resource to your collection using the `collection_id` you created earlier
    <CodeGroup>
  
```python Python
from wetro import Wetrocloud
    
# Access Module
client = Wetrocloud(api_key="your_api_key")
rag_client = client.rag
     
#Insert a Resource into a Collection
insert_response=rag_client.collection.insert_resource(
  collection_id="my_unique_collection_id",
  resource="https://medium.com/@wetrocloud/why-legal-tech-needs-wetrocloud-ai-rag-and-the-future-of-legal-practice-66fb38c4df09", 
  type="web"
)
    
print(insert_response)
```

```javascript Javascript
import Wetrocloud from "wetro-sdk";

const wetrocloud = new Wetrocloud();
const response = await wetrocloud.insertResource({
  collection_id: "<collection_id>",
  resource: "https://medium.com/@AlexanderObregon/a-brief-history-of-artificial-intelligence-1656693721f9#:~:text=In%20this%20article%2C%20we%20explore,learning%20are%20breaking%20new%20ground.",
  type: "web"
});

console.log("insert a resource", response);
````

```bash cURL
curl --request POST \
      --url https://api.wetrocloud.com/v1/resource/insert/ \
      --header 'Authorization: Token <api-key>' \
      --header 'Content-Type: multipart/form-data' \
      --form 'collection_id=<collection_id>' \
      --form 'resource=https://en.wikipedia.org/wiki/Elizabeth_II' \
      --form 'type=web'
```
    </CodeGroup>
  </Step>

  <Step title="Query Collection">
    Query your collection using the `collection_id` created earlier by providing a query in the `request_query` field.
    <CodeGroup>
```python Python
from wetro import Wetrocloud

# Access Module
client = Wetrocloud(api_key="your_api_key")
rag_client = client.rag

# Query the collection
query_response = rag_client.collection.query_collection(
  collection_id="my_unique_collection_id",
  request_query="What are the key points of the article?"
)

print(query_response)
```

```javascript Javascript
import Wetrocloud from "wetro-sdk";

const wetrocloud = new Wetrocloud();

const collectionId = '<collection_id>';
const query = 'What are the sales trends for Q1?';

const response = await wetrocloud.queryResource({
  collection_id: collectionId,
  request_query: query,
  stream: true
});

for await (const item of response) {
    console.log("Received item:", item);
}
```

```bash cURL
curl --request POST \
      --url https://api.wetrocloud.com/v1/collection/query/ \
      --header 'Authorization: Token <api-key>' \
      --header 'Content-Type: multipart/form-data' \
      --form 'collection_id=<collection_id>' \
      --form 'request_query=What is this all about?' \
```
    </CodeGroup> 
    <Info>
      To keep things simple we made use of the `/collection/query/` endpoint without `structured output`
    </Info>
  </Step>

  <Step title="Chat with Collection">
    Chat with your collection using the `collection_id` created earlier by providing a message in the `message` field and continue conversations by passing `chat_history`.
    <CodeGroup>
```python Python
from wetro import Wetrocloud
# Access Module
client = Wetrocloud(api_key="your_api_key")
rag_client = client.rag

# Create a chat history
chat_history = [
    {"role": "user", "content": "What is this collection about?"}, 
    {"role": "system", "content": "It stores research papers on AI technology."}
]

# Continue the conversation with context
chat_response = rag_client.collection.chat(
    collection_id="<collection_id>",
    message="Can you explain the latest paper's methodology?",
    chat_history=chat_history
)
print(chat_response)
``` 

```javascript JavaScript
import Wetrocloud from "wetro-sdk";

const wetrocloud = new Wetrocloud();
const response = await wetrocloud.chat({
    collection_id:"<collection_id>",
    message:"Tell me more",
    chat_history:[
        {"role":"user", "content":"What is this all about?"}, 
        {"role":"system","content":"This is about Queen Elizabeth_II of England"}
      ]
});

console.log("Chat with collection", response);
```
   
```bash cURL
   curl --request POST \
        --url https://api.wetrocloud.com/v1/collection/chat/ \
        --header 'Authorization: Token <api-key>' \
        --header 'Content-Type: multipart/form-data' \
        --form 'collection_id=<collection_id>' \
        --form 'message=Tell me more' \
        --form 'chat_history=[{"role":"user","content":"What is this all about?"}, {"role":"system","content":"This is about Queen Elizabeth_II of England"}]' \
```
    </CodeGroup>

  <Step title="Categorize a Resource">
    <CodeGroup>
  
 ```python Python
  from wetro import Wetrocloud

  client = Wetrocloud(api_key="your_api_key")
  tools_client = client.tools

  # Categorize content
  categorize_response = tools_client.categorize(
      resource="match review: John Cena vs. The Rock.",
      type="text",
      json_schema='{"label": "string"}',
      categories=["wrestling", "entertainment", "sports", "news"],
      prompt="Categorize the text to see which category it best fits")

  print(categorize_response)
  print("\nThe text is categorized as:", categorize_response.response["label"])
  ```
   
```javascript JavaScript

const wetrocloud = new Wetrocloud();

const response = await wetrocloud.categorize({
  resource: "match review: John Cena vs. The Rock",
  type: "text",
  json_schema: { 'label': '' },
  categories: ["football", "coding", "entertainment", "basketball", "wrestling", "information"],
  prompt: "Where does this fall under?"
});

console.log("Categorizing resource", result);
```

```bash cURL

        --form 'resource=match review: John Cena vs. The Rock.' \
```

  <Step title="Remove a Resource">
    This endpoint is used to delete a collection
    <CodeGroup>
  

```python Python
from wetro import Wetrocloud

client = Wetrocloud(api_key="your_api_key")
rag_client = client.rag

# Delete the resource from the collection
delete_response = rag_client.collection.delete_resource(
    collection_id="research_paper",
    resource_id=insert_response.resource_id
)
print(delete_response)
```

```javascript Javascript
import Wetrocloud from "wetro-sdk";

const wetrocloud = new Wetrocloud();
const response = await wetrocloud.deleteResource({
  collection_id: '<collection_id>',
  resource_id: '<resource_id>'
});

console.log("Deleting resource", response);
```

    ```bash cURL
        curl --request DELETE \
          --url https://api.wetrocloud.com/v1/resource/remove/ \
          --header 'Authorization: Token <api-key>'
          --form 'collection_id=<collection_id>' \
          --form 'resource_id=<resource_id>' \
      ```
    </CodeGroup>
  </Step>
