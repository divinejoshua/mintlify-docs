#### Introduction 
This guide is for developers looking to get started with the Wetrocloud API quickly. You’ll learn how to:

- Create a collection
- Insert resources into the collection
- Query collection for answers
- Chat with collection for conversations
- Categorize resources
- Remove resources
- Delete Collection

**Prerequisites:**
- An API key from the [Wetrocloud Console](https://console.wetrocloud.com)

By the end, you’ll have queried your first resource using Wetrocloud API.

  <Step title="Create a Collection">
    First, create a `collection_id` to group your resources. [Click Here to Learn more](/endpoint-explanations/create)
    <CodeGroup>
      ```python Python
    from wetro import Wetrocloud

    client = Wetrocloud(api_key="your_api_key")
    rag_client = client.rag
        
    # Set or create a collection
    rag_client.collection.create_collection("my_unique_collection_id")
      ```
      ```bash cURL
        curl --request POST \
          --url https://api.wetrocloud.com/v1/collection/create/ \
          --header 'Authorization: Token <api-key>'
          --form 'collection_id=<collection_id>' \
        ```
    
      ```javascript JavaScript
        const axios = require('axios');
        const FormData = require('form-data');

        const url = 'https://api.wetrocloud.com/v1/collection/create/';
        const headers = { 'Authorization': 'Token <api-key>' };
        const formData = new FormData();
        formData.append('collection_id', '<collection_id>');

        axios.post(url, formData, { headers })
          .then(response => console.log(response.data))
          .catch(error => console.error(error));
      ```
    </CodeGroup>
  </Step>

  <Step title="Insert a Resource into the Collection">
    Add a resource to your collection using the `collection_id` you created earlier  [Click Here to Learn more](/endpoint-explanations/insert)
    <CodeGroup>
      ```python Python
      from wetro import Wetrocloud

      client = Wetrocloud(api_key="your_api_key")
      rag_client = client.rag

      insert_response = rag_client.collection.insert(collection_id="my_unique_collection_id",resource="https://medium.com/@wetrocloud/why-legal-tech-needs-wetrocloud-ai-rag-and-the-future-of-legal-practice-66fb38c4df09", type="web")
      print("Insert response: %s", insert_response)
      ```
      ```bash cURL
        curl --request POST \
          --url https://api.wetrocloud.com/v1/resource/insert/ \
          --header 'Authorization: Token <api-key>' \
          --header 'Content-Type: multipart/form-data' \
          --form 'collection_id=<collection_id>' \
          --form 'resource=https://en.wikipedia.org/wiki/Elizabeth_II' \
          --form 'type=web'
      ```
      ```javascript JavaScript
        const axios = require('axios');
        const FormData = require('form-data');

        const url = 'https://api.wetrocloud.com/v1/resource/insert/';
        const formData = new FormData();
        formData.append('collection_id', '<collection_id>');
        formData.append('resource', 'https://en.wikipedia.org/wiki/Elizabeth_II');
        formData.append('type', 'web');

        const headers = {
            'Authorization': 'Token <api-key>',
            ...formData.getHeaders()
        };

        axios.post(url, formData, { headers })
            .then(response => console.log(response.data))
            .catch(error => console.error(error));
      ```
    </CodeGroup>
  </Step>

  <Step title="Query Collection">
    Query your collection using the `collection_id` created earlier by providing a query in the `request_query` field. [Click Here to Learn more](/endpoint-explanations/query)
    <CodeGroup>
      ```python Python
      from wetro import Wetrocloud

      client = Wetrocloud(api_key="your_api_key")
      rag_client = client.rag

      # Create collection.
      rag_client.collection.create_collection(collection_id="my_unique_collection_id")
        
      # Populate collection with resource. Note that you must have a resource in the collection to query.
      insert_response = rag_client.collection.insert(collection_id="my_unique_collection_id",resource="https://medium.com/@wetrocloud/why-legal-tech-needs-wetrocloud-ai-rag-and-the-future-of-legal-practice-66fb38c4df09", type="web")
      print("Insert response: %s", insert_response)

      # Query the populated collection
      query_response = rag_client.collection.query(collection_id="my_unique_collection_id", request_query="What are the key points of the article?")
      print("Query response: %s", query_response)
      ```
      ```bash cURL
        curl --request POST \
          --url https://api.wetrocloud.com/v1/collection/query/ \
          --header 'Authorization: Token <api-key>' \
          --header 'Content-Type: multipart/form-data' \
          --form 'collection_id=<collection_id>' \
          --form 'request_query=What is this all about?' \
      ``` 
      ```javascript JavaScript
        const axios = require('axios');
        const FormData = require('form-data');

        const url = 'https://api.wetrocloud.com/v1/collection/query/';
        const formData = new FormData();
        formData.append('collection_id', '<collection_id>');
        formData.append('request_query', 'What is this all about?');

        const headers = {
            'Authorization': 'Token <api-key>',
            ...formData.getHeaders()
        };

        axios.post(url, formData, { headers })
            .then(response => console.log(response.data))
            .catch(error => console.error(error));
      ```
    </CodeGroup> 
    <Info>
      To keep things simple we made use of the `/collection/query/` endpoint without `structured output`
    </Info>
  </Step>

  <Step title="Chat with Collection">
    Chat with your collection using the `collection_id` created earlier by providing a message in the `message` field and continue conversations by passing `chat_history`.
    <CodeGroup>
      ```python Python
      from wetro import Wetrocloud

      client = Wetrocloud(api_key="your_api_key")
      rag_client = client.rag
      
      # Create collection.
      rag_client.collection.create_collection(collection_id="my_unique_collection_id")
      
      # Populate collection with resource. Note that you must have a resource in the collection to query.
      insert_response = rag_client.collection.insert(collection_id="my_unique_collection_id",resource="https://medium.com/@wetrocloud/why-legal-tech-needs-wetrocloud-ai-rag-and-the-future-of-legal-practice-66fb38c4df09", type="web")
      print("Insert response: %s", insert_response)

      chat_history = [
          {"role": "user", "content": "What is this collection about?"}, 
          {"role": "system", "content": "It stores research papers on AI technology."}
      ]

      # Continue the conversation with context
      chat_response = rag_client.collection.chat(
          collection_id="my_unique_collection_id",
          message = "Can you explain the latest paper's methodology?",
          chat_history = chat_history
      )
      print(chat_response)
      ```
      ```bash cURL
        curl --request POST \
          --url https://api.wetrocloud.com/v1/collection/chat/ \
          --header 'Authorization: Token <api-key>' \
          --header 'Content-Type: multipart/form-data' \
          --form 'collection_id=<collection_id>' \
          --form 'message=Tell me more' \
          --form 'chat_history=[{"role":"user","content":"What is this all about?"}, {"role":"system","content":"This is about Queen Elizabeth_II of England"}]' \
      ```
      ```javascript JavaScript
        const axios = require('axios');
        const FormData = require('form-data');

        const url = 'https://api.wetrocloud.com/v1/collection/query/';
        const formData = new FormData();
        formData.append('collection_id', '<collection_id>');
        formData.append('message', 'Tell me more');
        formData.append('chat_history', [{"role":"user",  "content":"What is this all about?"}, 
            {"role":"system","content":"This is about Queen Elizabeth_II of England"}]);

        const headers = {
            'Authorization': 'Token <api-key>',
            ...formData.getHeaders()
        };

        axios.post(url, formData, { headers })
            .then(response => console.log(response.data))
            .catch(error => console.error(error));
      ```
    </CodeGroup> 
  </Step>

  <Step title="Categorize a Resource">
    <CodeGroup>
      ```python Python
      from wetro import Wetrocloud

      client = Wetrocloud(api_key="your_api_key")
      tools_client = client.tools

      # Categorize content
      categorize_response = tools_client.categorize(
        resource="match review: John Cena vs. The Rock.",
        type="text",
        json_schema='{"label": "string"}',
        categories=["wrestling", "entertainment", "sports", "news"],
        prompt="Categorize the text to see which category it best fits"
      )
      print(categorize_response)
      ```
      ```bash cURL
        curl --request POST \
          --url https://api.wetrocloud.com/v1/categorize/ \
          --header 'Authorization: Token <api-key>' \
          --header 'Content-Type: multipart/form-data' \
          --form 'resource=match review: John Cena vs. The Rock.' \
          --form type=text \
          --form 'json_schema={"label":"string" }' \
          --form 'categories=["football", "coding", "entertainment", "basketball", "wrestling", "information"]'
      ```
      ```javascript JavaScript
        const axios = require('axios');
        const FormData = require('form-data');

        const url = 'https://api.wetrocloud.com/v1/categorize/';
        const formData = new FormData();
        formData.append('resource', 'match review: John Cena vs. The Rock.');
        formData.append('type', 'text');
        formData.append('json_schema', JSON.stringify({"label": "string"}));
        formData.append('categories', JSON.stringify(["football", "coding", "entertainment", "basketball", "wrestling", "information"]));

        const headers = {
            'Authorization': 'Token <api-key>',
            ...formData.getHeaders()
        };

        axios.post(url, formData, { headers })
            .then(response => console.log(response.data))
            .catch(error => console.error(error));
      ```
    </CodeGroup>
  </Step>

  <Step title="Remove a Resource">
    This endpoint is used to delete a collection
    <CodeGroup>
      ```python Python
      from wetro import Wetrocloud

      client = Wetrocloud(api_key="your_api_key")
      rag_client = client.rag

      # Create collection.
      rag_client.collection.create_collection(collection_id="my_unique_collection_id")
      
      # Populate collection with resource. Note that you must have a resource in the collection to query.
      insert_response = rag_client.collection.insert(collection_id="my_unique_collection_id",resource="https://medium.com/@wetrocloud/why-legal-tech-needs-wetrocloud-ai-rag-and-the-future-of-legal-practice-66fb38c4df09", type="web")
      print("Insert response: %s", insert_response)

      # Delete a resource from the collection.
      delete_resource_resp = rag_client.collection.delete_resource(resource_id=insert_response.resource_id, collection_id="my_unique_collection_id")
      print("SDK Delete Resource Response: %s", delete_resource_resp)

      ```
      ```bash cURL
        curl --request DELETE \
          --url https://api.wetrocloud.com/v1/resource/remove/ \
          --header 'Authorization: Token <api-key>'
          --form 'collection_id=<collection_id>' \
          --form 'resource_id=<resource_id>' \
      ```
      ```javascript JavaScript
        const axios = require('axios');
        const FormData = require('form-data');

        const url = 'https://api.wetrocloud.com/v1/resource/remove/';
        const headers = {
            'Authorization': 'Token <api-key>'
        };
        const formData = new FormData();
        formData.append('collection_id', '<collection_id>');
        formData.append('resource_id', '<resource_id>');

        axios.delete(url, formData, { headers })
            .then(response => console.log(response.data))
            .catch(error => console.error(error));
      ```
    </CodeGroup>
  </Step>
